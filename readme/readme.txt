# user
- id
- name (string, позывной)
- real_name (string, настоящее имя)
- phone (string)
- socnet (string, адрес в соцсети)
- team_id (int, к какой команде записан)
- status (string, должность вус);
- ugroup (int, игрок, командир команды, участник ск, секретарь ск)

# team
- id
- name (string)
- status (int, БС ЧС СК)
- city (город базирования)
- description (text, о команде)
- timestamps

# camuflage
- id
- name (string, название камуфляжа)
- description (text, описание)
- team_id (правообладатель)

# game
- id
- name (string, название игры)
- location_name (string, населенный пункт и название полигона)
- location_coordinates (string, координаты полигона)
- description (text, Сценарий игры)
- team_id (int, команда-организатор)
- date_start (datetime, дата и время начала игры)
- date_end (datetime, дата и время конца игры)
